CREATE TABLE Login (
	LoginID INT PRIMARY KEY IDENTITY(1,1),
	Username VARCHAR(20),
	Password VARCHAR(20)
)

CREATE TABLE Customer (
	CustomerID INT PRIMARY KEY IDENTITY(1,1),
	FirstName VARCHAR(20),
	LastName VARCHAR(20)
	LoginID INT,
	FOREIGN KEY (LoginID) REFERENCES Login(LoginID)
)

CREATE TABLE Vehicle (
	VehicleID INT PRIMARY KEY,
	Make VARCHAR(40),
	Model VARCHAR(50),
	Year INT,
	VClass VARCHAR(40),
	City INT,
	Highway INT,
	Combo INT,
	FuelType VARCHAR(30),
	FuelCost INT,
	Drive VARCHAR(30),
	EngID INT,
	Eng_Desc VARCHAR(40),
	Cylinders INT,
	Liters DEC,
	Transmission VARCHAR(40),
	Trans_Desc VARCHAR(20)
)

CREATE TABLE CustomerVehicle (
	CVehicleID INT PRIMARY KEY IDENTITY(1,1),
	CVehicleMake VARCHAR(40),
	CVehicleModel VARCHAR(50),
	CVehicleYear INT,
	CVehicleFuelType VARCHAR(30),
	CVehicleDrive VARCHAR(30),
	CVehicleTrans VARCHAR(40),
	CVehicleCyl INT,
	CVehicleLit DEC,
	CVehicleOdometer INT,
	CustomerID INT,
	VehicleID INT,
	FOREIGN KEY (CustomerID) REFERENCES Customer(CustomerID),
	FOREIGN KEY (VehicleID) REFERENCES Vehicle(VehicleID)
)

CREATE TABLE Trip (
  TripID INT PRIMARY KEY IDENTITY(1,1),
  CVehicleID INT,
  VehicleID INT,
  PrevFillDate DATE, // Auto populate
  PrevFillLine DEC, // Auto populate
  PrevOdometer INT, // Auto populate from Previous Trip
  CurrOdometer INT,
  FillDate DATE,
  FillLine DEC, // Auto populate from manually chosen interval
  Distance VARCHAR, // Auto populate by (CurrOdometer - PrevOdometer)
  FOREIGN KEY (CVehicleID) REFERENCES CustomerVehicle(CVehicleID),
  FOREIGN KEY (VehicleID) REFERENCES Vehicle(VehicleID)
)
